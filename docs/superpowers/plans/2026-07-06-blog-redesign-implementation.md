# Blog Redesign Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a lightweight static blog publishing flow for `personal-site` using selected `../resources/*.md` files as the source.

**Architecture:** Keep the current static site model. Add a Node build script that scans knowledge-base resources, validates publish metadata, writes `generated/blog-index.json`, and generates static HTML under `blog/`. Then update the public pages so the site has home, about, blog, article, and cases entry points without adding a framework.

**Tech Stack:** Static HTML, CSS, vanilla JavaScript, Node.js built-in modules, Node test runner.

---

## File Structure

- Create `package.json`: npm scripts for build, test, and local static serving.
- Create `scripts/build-blog.js`: blog generator and exported pure functions for tests.
- Create `scripts/build-blog.test.js`: Node test runner coverage for metadata parsing, validation, and output generation.
- Modify selected `../resources/*.md`: add publish metadata to the first batch of public posts.
- Create `blog/index.html`: generated by the build script.
- Create `blog/<slug>/index.html`: generated article pages.
- Create `generated/blog-index.json`: generated blog metadata for home page cards and blog index behavior.
- Modify `index.html`: make the site a concise home page and add links to `/about/`, `/blog/`, and `/cases/`.
- Create `about/index.html`: static about page hydrated from `content/profile.md` and `content/resume.md`.
- Create `cases/index.html`: static cases page using existing case cards and Markdown content.
- Modify `script.js`: keep Markdown hydration and add generated latest-post rendering for the home page.
- Modify `styles.css`: add layout rules for home, about, blog, article, and cases pages.
- Modify `README.md`: document build and dev commands.

Generated artifacts under `blog/` and `generated/` should be committed because this site is static and may be deployed without a build pipeline.

```js
const fs = require("node:fs");
const path = require("node:path");

const REQUIRED_FIELDS = ["slug", "title", "summary", "category", "tags", "published"];

function escapeHtml(value) {
  return String(value)
    .replaceAll("&", "&amp;")
    .replaceAll("<", "&lt;")
    .replaceAll(">", "&gt;")
    .replaceAll('"', "&quot;");
}

function extractFrontmatter(markdown) {
  if (!markdown.startsWith("---\n")) {
    return { frontmatter: "", body: markdown };
  }

  const end = markdown.indexOf("\n---", 4);
  if (end === -1) {
    return { frontmatter: "", body: markdown };
  }

  const bodyStart = markdown.indexOf("\n", end + 4);
  return {
    frontmatter: markdown.slice(4, end).trim(),
    body: bodyStart === -1 ? "" : markdown.slice(bodyStart + 1).trim(),
  };
}

function parseScalar(value) {
  const trimmed = value.trim();

  if (trimmed === "true") return true;
  if (trimmed === "false") return false;

  if (trimmed.startsWith("[") && trimmed.endsWith("]")) {
    const inner = trimmed.slice(1, -1).trim();
    if (!inner) return [];
    return inner.split(",").map((item) => item.trim()).filter(Boolean);
  }

  return trimmed.replace(/^["']|["']$/g, "");
}

function parseFrontmatter(frontmatter) {
  const data = {};
  for (const rawLine of frontmatter.split(/\r?\n/)) {
    const line = rawLine.trim();
    if (!line || line.startsWith("#")) continue;

    const separator = line.indexOf(":");
    if (separator === -1) continue;

    const key = line.slice(0, separator).trim();
    const value = line.slice(separator + 1);
    data[key] = parseScalar(value);
  }
  return data;
}

function validateSlug(slug, filePath) {
  if (!/^[a-z0-9]+(?:-[a-z0-9]+)*$/.test(slug)) {
    throw new Error(`${filePath} has invalid slug: ${slug}`);
  }
}

function validatePost(post, filePath) {
  for (const field of REQUIRED_FIELDS) {
    if (post[field] === undefined || post[field] === "") {
      throw new Error(`${filePath} is missing required field: ${field}`);
    }
  }

  validateSlug(post.slug, filePath);

  if (!Array.isArray(post.tags)) {
    throw new Error(`${filePath} field tags must be an inline array`);
  }

  if (!/^\d{4}-\d{2}-\d{2}$/.test(post.published)) {
    throw new Error(`${filePath} field published must use YYYY-MM-DD`);
  }

  const timestamp = Date.parse(`${post.published}T00:00:00Z`);
  if (Number.isNaN(timestamp)) {
    throw new Error(`${filePath} field published is not a valid date`);
  }
}

function renderInline(value) {
  const escaped = escapeHtml(value);
  return escaped
    .replace(/\*\*([^*]+)\*\*/g, "<strong>$1</strong>")
    .replace(
      /(https?:\/\/[^\s<]+)/g,
      '<a href="$1" target="_blank" rel="noreferrer">$1</a>'
    );
}

function renderMarkdown(markdown) {
  const lines = markdown.split(/\r?\n/);
  const html = [];
  let listOpen = false;

  const closeList = () => {
    if (listOpen) {
      html.push("</ul>");
      listOpen = false;
    }
  };

  for (const rawLine of lines) {
    const line = rawLine.trim();

    if (!line) {
      closeList();
      continue;
    }

    if (line.startsWith("### ")) {
      closeList();
      html.push(`<h3>${renderInline(line.slice(4))}</h3>`);
      continue;
    }

    if (line.startsWith("## ")) {
      closeList();
      html.push(`<h2>${renderInline(line.slice(3))}</h2>`);
      continue;
    }

    if (line.startsWith("# ")) {
      closeList();
      html.push(`<h1>${renderInline(line.slice(2))}</h1>`);
      continue;
    }

    if (line.startsWith("- ")) {
      if (!listOpen) {
        html.push("<ul>");
        listOpen = true;
      }
      html.push(`<li>${renderInline(line.slice(2))}</li>`);
      continue;
    }

    closeList();
    html.push(`<p>${renderInline(line)}</p>`);
  }

  closeList();
  return html.join("\n");
}

function parseMarkdownFile(filePath) {
  const markdown = fs.readFileSync(filePath, "utf8");
  const { frontmatter, body } = extractFrontmatter(markdown);
  const metadata = parseFrontmatter(frontmatter);

  if (metadata.publish !== true) {
    return null;
  }

  const post = { ...metadata, body, sourcePath: filePath };
  validatePost(post, filePath);
  return post;
}

function pageShell({ title, description, active, body, relativePrefix = "./" }) {
  const nav = [
    ["首页", `${relativePrefix}index.html`, "home"],
    ["关于", `${relativePrefix}about/`, "about"],
    ["博客", `${relativePrefix}blog/`, "blog"],
    ["案例", `${relativePrefix}cases/`, "cases"],
  ];

  return `<!doctype html>
<html lang="zh-CN">
  <head>
    <meta charset="utf-8">
    <meta name="viewport" content="width=device-width, initial-scale=1">
    <title>${escapeHtml(title)}</title>
    <meta name="description" content="${escapeHtml(description)}">
    <link rel="stylesheet" href="${relativePrefix}styles.css">
  </head>
  <body>
    <header class="site-header">
      <a class="brand" href="${relativePrefix}index.html" aria-label="返回首页">夏目</a>
      <nav class="nav" aria-label="主导航">
        ${nav.map(([label, href, key]) => `<a href="${href}"${active === key ? ' aria-current="page"' : ""}>${label}</a>`).join("\n        ")}
      </nav>
    </header>
    ${body}
  </body>
</html>`;
}

function renderArticle(post) {
  const body = `<main class="article-layout">
      <article class="article">
        <a class="back-link" href="../">返回博客</a>
        <header class="article-header">
          <p class="eyebrow">${escapeHtml(post.category)}</p>
          <h1>${escapeHtml(post.title)}</h1>
          <p class="lead">${escapeHtml(post.summary)}</p>
          <div class="post-meta">
            <time datetime="${escapeHtml(post.published)}">${escapeHtml(post.published)}</time>
            <span>${post.tags.map(escapeHtml).join(" / ")}</span>
          </div>
        </header>
        <div class="markdown-panel article-body">
${renderMarkdown(post.body)}
        </div>
      </article>
      <aside class="author-footer">
        <h2>关于作者</h2>
        <p>夏目，关注 DevOps、AI 工作流、知识库和企业自动化工具落地。</p>
        <a class="button secondary" href="mailto:xb990815@163.com">联系交流</a>
      </aside>
    </main>`;

  return pageShell({
    title: `${post.title} | 夏目`,
    description: post.summary,
    active: "blog",
    body,
    relativePrefix: "../../",
  });
}

function renderBlogIndex(posts) {
  const cards = posts.map((post) => `<article class="post-card" data-category="${escapeHtml(post.category)}">
          <p class="eyebrow">${escapeHtml(post.category)}</p>
          <h2><a href="./${escapeHtml(post.slug)}/">${escapeHtml(post.title)}</a></h2>
          <p>${escapeHtml(post.summary)}</p>
          <div class="post-meta">
            <time datetime="${escapeHtml(post.published)}">${escapeHtml(post.published)}</time>
            <span>${post.tags.map(escapeHtml).join(" / ")}</span>
          </div>
        </article>`).join("\n");

  const categories = [...new Set(posts.map((post) => post.category))];
  const filters = ['<button class="filter-button active" type="button" data-category="all">全部</button>']
    .concat(categories.map((category) => `<button class="filter-button" type="button" data-category="${escapeHtml(category)}">${escapeHtml(category)}</button>`))
    .join("\n          ");

  const body = `<main class="page-layout">
      <section class="page-hero compact-hero">
        <p class="eyebrow">Blog</p>
        <h1>公开笔记与方法沉淀</h1>
        <p class="lead">这些文章来自知识库中明确标记为公开发布的 resources 内容。</p>
      </section>
      <section class="content-section single-column">
        <div class="filter-bar" aria-label="博客分类筛选">
          ${filters}
        </div>
        <div class="post-list" data-blog-list>
          ${cards || '<p class="empty-state">还没有已发布文章。</p>'}
        </div>
      </section>
    </main>
    <script src="../script.js"></script>`;

  return pageShell({
    title: "博客 | 夏目",
    description: "夏目的公开笔记、AI 工作流、知识库和自动化方法沉淀。",
    active: "blog",
    body,
    relativePrefix: "../",
  });
}

function ensureEmptyDir(dirPath) {
  fs.rmSync(dirPath, { recursive: true, force: true });
  fs.mkdirSync(dirPath, { recursive: true });
}

function buildBlog({ resourcesDir, siteDir }) {
  const files = fs
    .readdirSync(resourcesDir)
    .filter((file) => file.endsWith(".md"))
    .map((file) => path.join(resourcesDir, file));

  const posts = files
    .map(parseMarkdownFile)
    .filter(Boolean)
    .sort((a, b) => b.published.localeCompare(a.published));

  const seenSlugs = new Set();
  for (const post of posts) {
    if (seenSlugs.has(post.slug)) {
      throw new Error(`Duplicate post slug: ${post.slug}`);
    }
    seenSlugs.add(post.slug);
  }

  const generatedDir = path.join(siteDir, "generated");
  const blogDir = path.join(siteDir, "blog");
  ensureEmptyDir(generatedDir);
  ensureEmptyDir(blogDir);

  const indexData = posts.map((post) => ({
    slug: post.slug,
    title: post.title,
    summary: post.summary,
    category: post.category,
    tags: post.tags,
    published: post.published,
    url: `./${post.slug}/`,
  }));

  fs.writeFileSync(
    path.join(generatedDir, "blog-index.json"),
    `${JSON.stringify(indexData, null, 2)}\n`,
    "utf8"
  );

  fs.writeFileSync(path.join(blogDir, "index.html"), renderBlogIndex(posts), "utf8");

  for (const post of posts) {
    const postDir = path.join(blogDir, post.slug);
    fs.mkdirSync(postDir, { recursive: true });
    fs.writeFileSync(path.join(postDir, "index.html"), renderArticle(post), "utf8");
  }

  return posts;
}

if (require.main === module) {
  const siteDir = path.resolve(__dirname, "..");
  const resourcesDir = path.resolve(siteDir, "..", "resources");
  const posts = buildBlog({ resourcesDir, siteDir });
  console.log(`Generated ${posts.length} published blog post(s).`);
}

module.exports = {
  extractFrontmatter,
  parseFrontmatter,
  parseMarkdownFile,
  validatePost,
  buildBlog,
};
```

```js
function escapeHtml(value) {
  return String(value)
    .replaceAll("&", "&amp;")
    .replaceAll("<", "&lt;")
    .replaceAll(">", "&gt;");
}

function inlineMarkdown(value) {
  const escaped = escapeHtml(value);
  const linked = escaped.replace(
    /(https?:\/\/[^\s<]+)/g,
    '<a href="$1" target="_blank" rel="noreferrer">$1</a>'
  );
  return linked.replace(/\*\*([^*]+)\*\*/g, "<strong>$1</strong>");
}

function renderMarkdown(markdown) {
  const lines = markdown.split(/\r?\n/);
  const html = [];
  let listOpen = false;

  const closeList = () => {
    if (listOpen) {
      html.push("</ul>");
      listOpen = false;
    }
  };

  for (const rawLine of lines) {
    const line = rawLine.trim();

    if (!line) {
      closeList();
      continue;
    }

    if (line.startsWith("## ")) {
      closeList();
      html.push(`<h2>${inlineMarkdown(line.slice(3))}</h2>`);
      continue;
    }

    if (line.startsWith("# ")) {
      closeList();
      html.push(`<h1>${inlineMarkdown(line.slice(2))}</h1>`);
      continue;
    }

    if (line.startsWith("- ")) {
      if (!listOpen) {
        html.push("<ul>");
        listOpen = true;
      }
      html.push(`<li>${inlineMarkdown(line.slice(2))}</li>`);
      continue;
    }

    closeList();
    html.push(`<p>${inlineMarkdown(line)}</p>`);
  }

  closeList();
  return html.join("");
}

async function loadMarkdown(key) {
  try {
    const response = await fetch(files[key], { cache: "no-cache" });
    if (!response.ok) {
      throw new Error(`Failed to load ${files[key]}`);
    }
    return await response.text();
  } catch {
    return fallback[key] || "";
  }
}

async function loadBlogIndex() {
  try {
    const response = await fetch(`${basePath}generated/blog-index.json`, { cache: "no-cache" });
    if (!response.ok) {
      throw new Error("Failed to load blog index");
    }
    return await response.json();
  } catch {
    return [];
  }
}

async function hydrate() {
  const panels = document.querySelectorAll("[data-md]");
  for (const panel of panels) {
    const key = panel.getAttribute("data-md");
    const markdown = await loadMarkdown(key);
    panel.innerHTML = renderMarkdown(markdown);
  }
}

function renderPostCard(post) {
  const tags = Array.isArray(post.tags) ? post.tags.join(" / ") : "";
  return `<article class="post-card">
    <p class="eyebrow">${escapeHtml(post.category)}</p>
    <h3><a href="${basePath}blog/${escapeHtml(post.slug)}/">${escapeHtml(post.title)}</a></h3>
    <p>${escapeHtml(post.summary)}</p>
    <div class="post-meta">
      <time datetime="${escapeHtml(post.published)}">${escapeHtml(post.published)}</time>
      <span>${escapeHtml(tags)}</span>
    </div>
  </article>`;
}

async function hydrateLatestPosts() {
  const container = document.querySelector("[data-latest-posts]");
  if (!container) return;

  const posts = await loadBlogIndex();
  container.innerHTML = posts.slice(0, 3).map(renderPostCard).join("") ||
    '<p class="empty-state">还没有已发布文章。</p>';
}

function hydrateBlogFilters() {
  const list = document.querySelector("[data-blog-list]");
  const buttons = document.querySelectorAll("[data-category]");
  if (!list || buttons.length === 0) return;

  buttons.forEach((button) => {
    button.addEventListener("click", () => {
      const category = button.getAttribute("data-category");
      buttons.forEach((item) => item.classList.remove("active"));
      button.classList.add("active");

      list.querySelectorAll(".post-card").forEach((card) => {
        const visible = category === "all" || card.getAttribute("data-category") === category;
        card.hidden = !visible;
      });
    });
  });
}
```
